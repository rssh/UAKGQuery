\section{  Изменения в интерфейсных спецификациях }

\begin{itemize}
 \item 1.0.2 - 1.1.0 
   \begin{itemize}
     \item добавлена поддержка UNICODE строк.
     \item добавлена поддержка для BLOB.
     \item добавлен метод \verb|QueryManager::get_next_sequence_value|
     \item добавлен метод \verb|QueryManager::get_property|
   \end{itemize}
 \item 1.0.1 - 1.0.2 
   \begin{itemize}
     \item добавлены опции командной строки, управляющие 
            журналированием.
     \item Изменен порядок инициализации: добавлена функция
          библиотеки клиента: \verb|initUAKGQueryService|
    \end{itemize}
 \item 1.0.0 - 1.0.1 изменен IDL тег префикса.
\end{itemize}

\section{  Изменения реализационных характеристик }

\begin{itemize}
 \item 1.0.2 - 1.1.0 
 \begin{itemize}
   \item Добавлена поддержка InterBase-6.0 
   \item Используется версия 1.4.0 GradSoft C++ ToolBox
   \item Исправлены известные ошибки версии 1.0.2
   \item Прекращена поддержка ORBacus-3.x и OmniORB-3.x
 \end{itemize}

 \item 1.0.1 - 1.0.2
 \begin{itemize}
   \item Совместим с версией 1.2.0 GradSoft C++ ToolBox.
   \item Изменена схема сообщений об ошибках: сейчас используется Logger.
   \item Убран вывод сообщения об ошибке инициализации, когда UAKGQueryService
   запущен с опцией \verb|--help|.
 \end{itemize}
 \item 1.0.0 - 1.0.1
 \begin{itemize}
   \item Исправлена ошибка в RC-кодировании целых данных на машинах с
 прямым порядком байт в слове.
   \item В отсуствие текущей транзакции изменения фиксируются непосредственно
  после выполнения SQL предложения в транзакционном режиме.
  \item Небольшие изменения в Руководстве Программиста.
  \item Исправлена ошибка в процессе инсталляции, когда включаемые файлы
   GradSoft C++ ToolBox  находятся в директории, которая не указана как
   источник включаемых файлоа для компилятора С++ по умолчанию.
  \item Исправленна ошибка в "Руководстве Администратора" о порядке инсталляции
    по Windows NT
 \end{itemize}
\end{itemize}
